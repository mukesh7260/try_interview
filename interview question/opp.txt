import asyncio
async def factorial(n):
    result = 1
    for i in range(1, n + 1):
        result *= i
        await asyncio.sleep(0.3)  # Simulate some work
    print(f"Factorial of {n}: {result}")

async def reverse_string(string):
    reversed_str = string[::-1]
    await asyncio.sleep(0.2)  # Simulate some work
    print(f"Reversed string: {reversed_str}")

async def fibonacci(n):
    fib_sequence = [0, 1]
    for i in range(2, n):
        next_fib = fib_sequence[-1] + fib_sequence[-2]
        fib_sequence.append(next_fib)
        await asyncio.sleep(0.6)  # Simulate some work
    print(f"Fibonacci sequence: {fib_sequence}")

async def main():
    tasks = [
        factorial(int(input('enter any number '))),
        reverse_string(input('enter any string ')),
        fibonacci(int(input('how many no. of fabonaccies series  ')))
    ]

    await asyncio.gather(*tasks)
asyncio.run(main())


			3. Async program one by one 

import asyncio

async def factorial(n):
    result = 1
    for i in range(1, n + 1):
        result *= i
        await asyncio.sleep(0.1)  # Simulate some async work
    return result

async def reverse_string(s):
    await asyncio.sleep(0.1)  # Simulate some async work
    return s[::-1]

async def fibonacci(n):
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    sequence = [0, 1]
    for i in range(2, n):
        sequence.append(sequence[i - 1] + sequence[i - 2])
        await asyncio.sleep(0.1)  # Simulate some async work
    return sequence

async def main():
    tasks = [
        asyncio.create_task(factorial(5)),
        asyncio.create_task(reverse_string("hello")),
        asyncio.create_task(fibonacci(10))
    ]

    results = await asyncio.gather(*tasks)
    print("Factorial:", results[0])
    print("Reversed String:", results[1])
    print("Fibonacci Sequence:", results[2])

asyncio.run(main())

def merge(s1, s2):	
	result = ""	
	i = 0
	while (i < len(s1)) or (i < len(s2)):	
		if (i < len(s1)):
			result += s1[i]
		if (i < len(s2)):
			result += s2[i]	
		i += 1	
	return result

How To Use to_representation
Serializers provide a hook point, called to_representation.

Suppose we want to add a key called admin to serialized data, only when the user is a superuser. For non-superusers, the key admin shouldnâ€™t be present in serialized data.

Our serializer would look like:

class UserSerializer(serializers.ModelSerializer):

    class Meta:
        model = User
        fields = ('email', 'first_name', 'last_name')

    def to_representation(self, instance):
        representation = super().to_representation(instance)
        if instance.is_superuser:
            representation['admin'] = True
        return representation

        return []

def permutations(s):  
    if len(s)==0:           # if string is empty 
        return [] 
    if len(s)==1:
        return [s]    
    
    per = []        # store the permutations in the list . 

    for i in range(len(s)):      # Iterate over each character in the string
        current_char = s[i]               # Get the current character
        remaining_string = s[:i] + s[i + 1:]                # Generate the remaining string by excluding the current character
        sub_perms = permutations(remaining_string)      # Recursively generate permutations of the remaining string
        for sub_perm in sub_perms:
            per.append(current_char + sub_perm)         # Append the current character to each permutation of the remaining string
    return per
    
input_string = 'ABC'
permutations = permutations(input_string)
print(permutations)

# views.py
from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework import status
from rest_framework.permissions import IsAuthenticated
from rest_framework_simplejwt.tokens import RefreshToken
from .models import CouplesPhoto
from .serializers import CouplesPhotoSerializer

class CouplesPhotoListCreateAPIView(APIView):
    permission_classes = [IsAuthenticated]

    def get(self, request):
        couples_photos = CouplesPhoto.objects.all()
        serializer = CouplesPhotoSerializer(couples_photos, many=True)
        return Response(serializer.data)

    def post(self, request):
        serializer = CouplesPhotoSerializer(data=request.data)
        if serializer.is_valid():
            serializer.save()
            return Response(serializer.data, status=status.HTTP_201_CREATED)
        return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

class CouplesPhotoDetailAPIView(APIView):
    permission_classes = [IsAuthenticated]

    def get_object(self, pk):
        try:
            return CouplesPhoto.objects.get(pk=pk)
        except CouplesPhoto.DoesNotExist:
            return None

    def get(self, request, pk):
        couples_photo = self.get_object(pk)
        if couples_photo:
            serializer = CouplesPhotoSerializer(couples_photo)
            return Response(serializer.data)
        else:
            return Response(status=status.HTTP_404_NOT_FOUND)

    def put(self, request, pk):
        couples_photo = self.get_object(pk)
        if couples_photo:
            serializer = CouplesPhotoSerializer(couples_photo, data=request.data)
            if serializer.is_valid():
                serializer.save()
                return Response(serializer.data)
            return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
        else:
            return Response(status=status.HTTP_404_NOT_FOUND)

    def delete(self, request, pk):
        couples_photo = self.get_object(pk)
        if couples_photo:
            couples_photo.delete()
            return Response(status=status.HTTP_204_NO_CONTENT)
        else:
            return Response(status=status.HTTP_404_NOT_FOUND)

class TokenObtainPairAPIView(APIView):
    def post(self, request):
        username = request.data.get('username')
        password = request.data.get('password')

        if username and password:
            user = authenticate(request, username=username, password=password)
            if user:
                refresh = RefreshToken.for_user(user)
                return Response({
                    'access': str(refresh.access_token),
                    'refresh': str(refresh)
                }, status=status.HTTP_200_OK)
            else:
                return Response({'error': 'Invalid username or password'}, status=status.HTTP_401_UNAUTHORIZED)
        else:
            return Response({'error': 'Username and password are required'}, status=status.HTTP_400_BAD_REQUEST)
