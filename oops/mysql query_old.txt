			FOREIGN KEY . 

A foreign key is a key used to link two tables together . 
A foreign key is a field (or collection of fields) in one table refers to the primary key in another table.

create table customers(cid int not null auto_increment primary key, cname varchar(55), email varchar(44));  

create table orders(oid int not null auto_increment primary key,orderdate date , cid int, foreign key(cid) references customers(cid)); 

			SQL JOINING
select * from customers,orders where customers.cid=orders.cid;   ## adding two table. 

select * from customers inner join orders ON customers.cid = orders.cid;    -- inner  join.   

select * from customers left join orders ON customers.cid = orders.cid;     -- left join.            
    
select * from customers right join orders ON customers.cid = orders.cid;     -- right join.   

	here  customers = 1st table, orders = 2nd table.
